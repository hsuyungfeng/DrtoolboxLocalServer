from src.services.escalation_handler import EscalationHandler

def route_message(user_id, text):
    rag_response = query_rag(text)
    if rag_response.confidence < 0.60:
        escalation = EscalationHandler().create_escalation(
            patient_id=user_id,
            original_message=text,
            rag_confidence=rag_response.confidence,
            conversation_history=ConversationManager().get_conversation_history(user_id, days=7)
        )
        return {"routing_decision": "escalation", "escalation_id": escalation.id}
